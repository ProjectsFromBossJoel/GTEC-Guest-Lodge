        function openWAModal() {
            const b = _lastBooking;

            // Populate fields from last booking
            document.getElementById('wa-display-id').textContent       = b.bookingId   || '—';
            document.getElementById('wa-display-name').textContent     = b.guestName   || '—';
            document.getElementById('wa-display-room').textContent     = b.roomNumber  || '—';

            document.getElementById('wa-display-checkin').textContent  = b.checkIn    || '—';
            document.getElementById('wa-display-checkout').textContent = b.checkOut   || '—';

            // Pre-fill phone if we have it
            if (b.guestPhone) {
                let p = b.guestPhone.replace(/\D/g, '');
                if (p.startsWith('233')) p = p.slice(3);
                if (p.startsWith('0'))   p = p.slice(1);
                document.getElementById('wa-phone-input').value = p;
            } else {
                document.getElementById('wa-phone-input').value = '';
            }

            // Build preview text
            const msgHTML = buildWAPreview(b);
            document.getElementById('wa-msg-preview-text').innerHTML = msgHTML;
            document.getElementById('wa-msg-time').textContent =
                new Date().toLocaleTimeString('en-GB', { hour: '2-digit', minute: '2-digit' });

            // Reset to main state
            document.getElementById('wa-main-content').style.display = '';
            document.getElementById('wa-sent-state').style.display   = 'none';

            closeSuccessOverlay();

            document.body.style.overflow = 'hidden';

        function waSend() {
            const raw = document.getElementById('wa-phone-input').value.trim();
            if (!raw || raw.length < 9) {
                waToast('⚠️ Enter a valid WhatsApp number');
                document.getElementById('wa-phone-input').focus();
                return;
            }

            let digits = raw.replace(/\D/g, '');
            if (digits.startsWith('0'))   digits = digits.slice(1);
            if (digits.startsWith('233')) digits = digits.slice(3);
            const fullNumber = '233' + digits;

            const message = buildWARaw(_lastBooking);
            const url     = `https://wa.me/${fullNumber}?text=${encodeURIComponent(message)}`;
            window.open(url, '_blank');

            // Show sent confirmation
            document.getElementById('wa-main-content').style.display = 'none';
            document.getElementById('wa-sent-state').style.display   = 'flex';
        }

        function openBookingModal(roomId, roomNumber, roomType) {

        function showToast(msg, success = true) {
            const t = document.createElement('div');
            t.className = 'toast';
            t.setAttribute('role', 'status');
            t.setAttribute('aria-live', 'polite');
            t.style.background = success ? '#0d1117' : '#c0392b';
            t.innerHTML = msg;
            document.body.appendChild(t);
            setTimeout(() => t.classList.add('show'), 100);
            setTimeout(() => { t.classList.remove('show'); setTimeout(() => t.remove(), 400); }, 3500);
        }
